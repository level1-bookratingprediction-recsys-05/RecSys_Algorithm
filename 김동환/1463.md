```
import math
dp = [10**6 for i in range(n+1)]
n = int(input())
dp[1] = 0

for i in range(1, n+1):
    if i % 3 == 0:
        cost = min(dp[i], dp[i//3] + 1)
        dp[i] = cost
        
    if i % 2 == 0:
        cost = min(dp[i], dp[i//2] + 1)
        dp[i] = cost

    cost = min(dp[i], dp[i-1] + 1)

    dp[i] = cost
print(dp[n])
